#include <bits/stdc++.h>
#define ll long long
using namespace std;
struct node
{
    ll value,b;
};
bool cmp(node x,node y)
{
    return x.value < y.value;
}
void Solve()
{
    ll n,k,ans = 0;
    cin >> n >> k;
    node a[n+1];
    for (int i = 1; i <= n; i++) cin >> a[i].value;
    for (int i = 1; i <= n; i++) cin >> a[i].b;
    sort(a+1,a+n+1,cmp);
    if (a[n].b == 1)
    {
        cout << a[n].value + a[n/2].value + k << "\n";
        return;
    }
    int p = 0;
    for (int i = n; i >= 1; i--)
    {
        if (a[i].b == 1)
        {
            p = i;
            break;
        }
    }
    ans = max(ans,a[n].value + a[n/2].value);
    if (p == 0)
    {
        cout << ans << "\n";
        return;
    }
    ll l = a[n/2].value,r = l + k;
    while (l <= r)
    {
        ll mid = (l + r) >> 1;
        ll used = 0;
        ll all = n;
        for (int i = n; i >= 1; i--)
        {
            if (a[i].value >= mid) all--;
            else
            {
                if (a[i].b == 1 && used + mid - a[i].value <= k)
                {
                    used += (mid - a[i].value);
                    all--;
                }
            }
        }
        if (all < n/2)
        {
            ans = max(ans,mid + a[n].value);
            l = mid + 1;
        }
        else r = mid - 1;
    }
    a[p].value += k;
    sort(a+1,a+n+1,cmp);
    ans = max(ans,a[n].value + a[n/2].value);
    cout << ans << "\n";
}
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int t = 1;
    cin >> t;
    while (t--)
    {
        Solve();
    }
    return 0;
}
